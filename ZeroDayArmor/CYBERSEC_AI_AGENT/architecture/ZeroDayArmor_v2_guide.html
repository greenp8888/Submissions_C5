Key Architecture Decisions Explained

Why sync nodes, not async?

LangGraph's .stream() and .invoke() are synchronous. Async nodes require asyncio.run() wrapping at every call site. For a hackathon and early production, sync nodes are simpler, fully debuggable, and work identically with all checkpointers. Async can be added later by wrapping node calls in asyncio.to_thread().

Why update_state over Command?

Command(resume=...) is designed for use inside interrupt() — a mid-node pause. interrupt_before pauses the graph before the node executes. The correct resume pattern is: inject state with update_state(), then call invoke(None, config) to re-enter the graph at the paused node with the updated state visible.

Why Annotated[List, operator.add]?

In LangGraph, if two nodes update the same state key, the last write wins by default. For lists like log_alerts, you want accumulation across nodes in the same cycle. The Annotated reducer tells LangGraph to extend the list rather than replace it — critical for collecting alerts from multiple parallel agents.

# WRONG (v1): app.invoke(Command(resume={"human_decision": "approved"}), config)

def resume_with_decision(app, thread_id: str, decision: str) -> dict:
    config = {"configurable": {"thread_id": thread_id}}
    # Step 1: inject the human decision into the paused state
    app.update_state(config, {"human_decision": decision})
    # Step 2: resume — invoke with None means "continue from checkpoint"
    result = {}
    for event in app.stream(None, config=config, stream_mode="updates"):
        result.update(list(event.values())[0])
    return result

from langchain_core.messages import HumanMessage, SystemMessage
from langsmith import traceable
from pydantic import BaseModel

class BaseSecurityAgent:
    # B-07 fix: .with_structured_output() replaces deprecated PydanticOutputParser
    # B-09 fix: no longer tied to OpenAI function calling — works with Gemini/Groq
    agent_name: str = "base"
    system_prompt: str = "You are a cybersecurity expert."
    output_schema = None

    def __init__(self):
        self.llm = settings.get_llm()

    @traceable(run_type="chain")
    def _invoke_llm(self, user_content: str):
        """Provider-agnostic LLM call with structured output."""
        messages = [
            SystemMessage(content=self.system_prompt),
            HumanMessage(content=user_content),
        ]
        llm = (self.llm.with_structured_output(self.output_schema)
               if self.output_schema else self.llm)
        return llm.invoke(messages)

    def run(self, state: dict) -> dict:
        """LangGraph node entry — calls _process(), handles errors."""
        try:
            return self._process(state)
        except Exception as e:
            return {"final_summary": f"[{self.agent_name}] ERROR: {e}",
                    "severity_level": state.get("severity_level", "CLEAR")}

    def _process(self, state: dict) -> dict:
        raise NotImplementedError

PYTHONorchestrator.py
from langgraph.graph import StateGraph, END
from typing import TypedDict, List, Any
from agents import LogMonitorAgent, ThreatIntelAgent, VulnScannerAgent
from agents import IncidentResponseAgent, PolicyCheckerAgent

class SecurityState(TypedDict):
    input_data: str
    input_type: str          # log | cve_query | code | config | incident
    log_alerts: List[Any]
    threat_reports: List[Any]
    scan_results: List[Any]
    playbooks: List[Any]
    compliance_reports: List[Any]
    final_summary: str
    severity_level: str      # aggregated max severity

def route_input(state: SecurityState) -> str:
    """Router: decides which agent runs based on input_type."""
    t = state["input_type"]
    if t == "log":     return "log_monitor"
    if t == "cve_query": return "threat_intel"
    if t == "code":     return "vuln_scanner"
    if t == "config":   return "policy_checker"
    if t == "incident": return "incident_response"
    return "log_monitor"  # default

def should_escalate(state: SecurityState) -> str:
    """After log monitor: escalate CRITICAL to incident response."""
    for alert in state.get("log_alerts", []):
        if alert.get("severity") == "CRITICAL":
            return "incident_response"
    return END

def build_graph() -> StateGraph:
    graph = StateGraph(SecurityState)

    # Add nodes
    graph.add_node("log_monitor",       LogMonitorAgent().run)
    graph.add_node("threat_intel",      ThreatIntelAgent().run)
    graph.add_node("vuln_scanner",      VulnScannerAgent().run)
    graph.add_node("incident_response", IncidentResponseAgent().run)
    graph.add_node("policy_checker",    PolicyCheckerAgent().run)

    # Entry + routing
    graph.set_entry_point("router")
    graph.add_node("router", lambda s: s)  # passthrough
    graph.add_conditional_edges("router", route_input, {
        "log_monitor":       "log_monitor",
        "threat_intel":      "threat_intel",
        "vuln_scanner":      "vuln_scanner",
        "policy_checker":    "policy_checker",
        "incident_response": "incident_response",
    })

    # Log monitor can auto-escalate
    graph.add_conditional_edges("log_monitor", should_escalate, {
        "incident_response": "incident_response",
        END: END,
    })

    # All others terminate
    graph.add_edge("threat_intel",      END)
    graph.add_edge("vuln_scanner",      END)
    graph.add_edge("incident_response", END)
    graph.add_edge("policy_checker",    END)

    return graph.compile()

# Usage
if __name__ == "__main__":
    app = build_graph()
    result = app.invoke({
        "input_type": "log",
        "input_data": open("data/sample_logs/auth.log").read()
    })
    print(result["final_summary"])

from langchain.agents import create_openai_tools_agent, AgentExecutor
from langchain_openai import ChatOpenAI
from langchain.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain.output_parsers import PydanticOutputParser
from tools.log_tools import regex_parser_tool, ip_reputation_tool, geo_lookup_tool
from schemas.outputs import LogAlert

SYSTEM_PROMPT = """You are a senior SOC analyst with 10+ years experience.
Analyze log entries and detect security threats. Use your tools to:
1. Parse logs with regex_parser_tool to find attack patterns
2. Check suspicious IPs with ip_reputation_tool
3. Look up geo-location with geo_lookup_tool for context
4. Return structured JSON matching LogAlert schema

Severity rules:
- CRITICAL: active RCE, data exfiltration in progress, ransomware
- HIGH: brute force success, privilege escalation, lateral movement
- MEDIUM: reconnaissance, failed brute force (>10 attempts), scanning
- LOW: single failed login, unusual hours access, new user agent

Always include MITRE ATT&CK technique IDs (e.g., T1110.001 for brute force).
Format: {format_instructions}"""

class LogMonitorAgent:
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        self.parser = PydanticOutputParser(pydantic_object=LogAlert)
        self.tools = [regex_parser_tool, ip_reputation_tool, geo_lookup_tool]

        prompt = ChatPromptTemplate.from_messages([
            ("system", SYSTEM_PROMPT.format(
                format_instructions=self.parser.get_format_instructions()
            )),
            ("human", "Analyze these logs:\n{input}"),
            MessagesPlaceholder("agent_scratchpad"),
        ])

        agent = create_openai_tools_agent(self.llm, self.tools, prompt)
        self.executor = AgentExecutor(agent=agent, tools=self.tools, verbose=True)

    def run(self, state: dict) -> dict:
        result = self.executor.invoke({"input": state["input_data"]})
        alert = self.parser.parse(result["output"])
        state["log_alerts"] = [alert.model_dump()]
        state["severity_level"] = alert.severity.value
        return state

🔁 Autonomous Self-Heal Loop

How the Verify step confirms healing worked, persists state across crashes, gates destructive actions behind human approval, and restarts the continuous monitoring cycle — all inside a LangGraph cyclic StateGraph.

LangGraph Cycles SqliteSaver Checkpointer interrupt_before Gates Auto-Remediation
ℹ️ The original architecture runs once and exits (DAG). The self-heal upgrade turns it into a persistent cyclic graph that monitors forever, heals automatically, and resumes after crashes — using three LangGraph features the previous design didn't use at all.

What the Verify Step Actually Does

1. Proof of Healing

After any auto-heal action (firewall rule added, service restarted, IP blocked), the Verify agent re-runs the exact check that triggered the original alert and compares current state against the pre-incident baseline. If the threat indicator is still present, it escalates rather than silently looping.

2. Incident Closure

On confirmed healing, the agent writes a closure record to SQLite with: incident ID, duration, actions taken, CVSS score at detection, and post-heal score. This feeds the LangSmith evaluation loop for continuous improvement of remediation quality.

3. Loop Restart

The graph routes from verify back to supervisor via a conditional edge. The SecurityState is reset (alerts cleared, counters incremented) but the monitoring baseline is updated with new learned patterns. The loop restarts within 2 seconds.

from langgraph.graph import StateGraph, END
from langgraph.checkpoint.sqlite import SqliteSaver
from langgraph.types import Command
from typing import TypedDict, List, Any, Annotated
import operator, time, uuid

# ── Cyclic State (never resets between loops) ──────────────
class SecurityState(TypedDict):
    input_data: str
    input_type: str
    log_alerts: Annotated[List[Any], operator.add]   # accumulates
    threat_reports: Annotated[List[Any], operator.add]
    scan_results: Annotated[List[Any], operator.add]
    playbooks: Annotated[List[Any], operator.add]
    compliance_reports: Annotated[List[Any], operator.add]
    severity_level: str
    heal_actions_taken: Annotated[List[str], operator.add]
    heal_verified: bool
    heal_attempts: int
    incident_id: str
    loop_count: int                                   # total monitoring cycles
    human_decision: str                               # "approved" | "rejected" | ""
    final_summary: str

# ── Severity Router ────────────────────────────────────────
def route_by_severity(state: SecurityState) -> str:
    sev = state.get("severity_level", "LOW")
    if sev in ("HIGH", "CRITICAL"):
        return "human_approval"    # interrupt_before fires here
    elif sev == "MEDIUM":
        return "auto_heal"
    else:
        return "verify"             # LOW → just verify + loop

# ── Auto-Heal Node ─────────────────────────────────────────
async def auto_heal_node(state: SecurityState) -> dict:
    actions = []
    for alert in state.get("log_alerts", []):
        alert_type = alert.get("alert_type")
        src_ip = alert.get("source_ip", "")

        if alert_type == "brute_force":
            actions += [
                f"iptables -A INPUT -s {src_ip} -j DROP",
                f"fail2ban-client set sshd banip {src_ip}",
                "systemctl restart sshd",
            ]
        elif alert_type == "rce":
            actions += [
                f"iptables -A INPUT -s {src_ip} -j DROP",
                "systemctl stop nginx",
                "snapshot --volume /var/www --tag pre-incident",
            ]
        elif alert_type == "exfil":
            actions += [
                f"iptables -A OUTPUT -d {src_ip} -j DROP",
                "tcpdump -i eth0 -w /tmp/incident_capture.pcap &",
            ]
        elif alert_type == "cryptomining":
            actions += [
                "kubectl delete pod -l app=suspicious --force",
                "kubectl apply -f network-policy-block-mining.yaml",
            ]

    # In production: execute via subprocess with timeout + dry-run flag
    # for cmd in actions: subprocess.run(cmd, shell=True, timeout=30)

    return {
        "heal_actions_taken": actions,
        "heal_attempts": state.get("heal_attempts", 0) + 1,
        "incident_id": state.get("incident_id") or str(uuid.uuid4()),
    }

# ── Human Approval Node (paused by interrupt_before) ───────
async def human_approval_node(state: SecurityState) -> dict:
    """
    Graph PAUSES before this node due to interrupt_before=["human_approval"].
    Human reviews via Slack webhook / API / CLI then calls:
      app.invoke(Command(resume={"human_decision": "approved"}), config=config)
    This node then executes with the decision in state.
    """
    decision = state.get("human_decision", "")
    if decision == "approved":
        return await auto_heal_node(state)
    else:
        return {
            "heal_actions_taken": ["Escalated to SOC — human rejected auto-heal"],
            "final_summary": "Incident pending manual investigation",
        }

# ── Verify Node ────────────────────────────────────────────
async def verify_node(state: SecurityState) -> dict:
    alerts = state.get("log_alerts", [])
    actions = state.get("heal_actions_taken", [])

    if not alerts:
        return {"heal_verified": True, "loop_count": state.get("loop_count", 0) + 1}

    # Re-check: call the same detection tool that fired the alert
    alert_type = alerts[-1].get("alert_type")
    src_ip = alerts[-1].get("source_ip", "")

    heal_confirmed = False
    if alert_type == "brute_force":
        # Check: is the IP now blocked in iptables?
        # import subprocess
        # result = subprocess.run(f"iptables -L -n | grep {src_ip}", ...)
        # heal_confirmed = src_ip in result.stdout
        heal_confirmed = any(src_ip in a for a in actions)

    elif alert_type == "rce":
        # Check: is the vulnerable service stopped?
        heal_confirmed = any("stop" in a or "snapshot" in a for a in actions)

    elif alert_type in ("exfil", "cryptomining"):
        heal_confirmed = len(actions) > 0

    if not heal_confirmed and state.get("heal_attempts", 0) < 3:
        # Retry with escalated actions
        return {"heal_verified": False, "severity_level": "CRITICAL"}

    return {
        "heal_verified": heal_confirmed,
        "loop_count": state.get("loop_count", 0) + 1,
        "log_alerts": [],          # clear resolved alerts
        "heal_actions_taken": [],   # clear for next cycle
        "severity_level": "CLEAR",
        "final_summary": f"Incident {state.get('incident_id','?')} closed after {state.get('heal_attempts',0)} heal attempt(s)",
    }

# ── After Verify: route back to monitoring loop ─────────────
def route_after_verify(state: SecurityState) -> str:
    if state.get("heal_verified"):
        return "supervisor"      # healed — restart monitoring loop
    elif state.get("heal_attempts", 0) >= 3:
        return END               # give up after 3 attempts — human must take over
    else:
        return "human_approval" # escalate unresolved incident

# ── Build Cyclic Graph ─────────────────────────────────────
def build_selfheal_graph():
    # Persist state to SQLite — survives process restarts
    checkpointer = SqliteSaver.from_conn_string("./cybersec_state.db")

    graph = StateGraph(SecurityState)

    # Add all nodes
    graph.add_node("supervisor",      supervisor_node)
    graph.add_node("log_monitor",     LogMonitorAgent().run)
    graph.add_node("threat_intel",    ThreatIntelAgent().run)
    graph.add_node("vuln_scanner",    VulnScannerAgent().run)
    graph.add_node("auto_heal",       auto_heal_node)
    graph.add_node("human_approval",  human_approval_node)
    graph.add_node("verify",          verify_node)

    graph.set_entry_point("supervisor")

    # Parallel agent execution via Send API
    graph.add_conditional_edges("supervisor", route_to_agents)
    graph.add_conditional_edges("log_monitor", route_by_severity, {
        "human_approval": "human_approval",
        "auto_heal":      "auto_heal",
        "verify":         "verify",
    })
    graph.add_edge("auto_heal",      "verify")
    graph.add_edge("human_approval", "verify")
    graph.add_conditional_edges("verify", route_after_verify, {
        "supervisor":      "supervisor",  # ← THE LOOP
        "human_approval":  "human_approval",
        END: END,
    })

    return graph.compile(
        checkpointer=checkpointer,
        interrupt_before=["human_approval"],   # pause for destructive actions
    )

# ── Running the perpetual loop ─────────────────────────────
if __name__ == "__main__":

def severity_accuracy(run, example) -> EvaluationResult:
    """Check predicted severity matches ground-truth label."""
    predicted = run.outputs.get("severity_level", "")
    expected = example.outputs.get("expected_severity", "")
    score = 1.0 if predicted == expected else 0.0
    return EvaluationResult(key="severity_accuracy", score=score,

def mitre_coverage(run, example) -> EvaluationResult:
    """Check that MITRE ATT&CK technique is correctly identified."""
    alerts = run.outputs.get("log_alerts", [])
    if not alerts: return EvaluationResult(key="mitre_coverage", score=0.0)
    technique = alerts[0].get("mitre_technique", "")
    expected = example.outputs.get("expected_mitre", "")
    score = 1.0 if expected in technique else 0.0
    return EvaluationResult(key="mitre_coverage", score=score)

Feedback scores: use thumbs up/down on LangSmith runs to build eval set

Alerts to Set Up in LangSmith

P95 latency > 15s for any agent → check LLM API status
Severity accuracy < 90% → review prompts and few-shot examples
Heal quality score < 0.7 → add more remediation examples to RAG
Daily cost report: tokens × model price = $X/day

from langsmith import Client
from langsmith.wrappers import wrap_openai
from openai import OpenAI

# Wrap OpenAI client — all calls automatically traced
openai_client = wrap_openai(OpenAI())

# Compare two prompt versions in LangSmith A/B test
def test_prompt_v1(log_entry: str) -> str:
    resp = openai_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content": "You are a SOC analyst. Classify this log entry."},
            {"role": "user", "content": log_entry},
        ],
    )
    return resp.choices[0].message.content

def test_prompt_v2(log_entry: str) -> str:
    """Improved: includes MITRE context + structured output instruction."""
    resp = openai_client.chat.completions.create(
        model="gpt-4o-mini",

Classify this log entry as JSON: {alert_type, severity, mitre_technique, confidence_score}.
Severity scale: CRITICAL=active_rce/exfil, HIGH=brute_force_success/scan, MEDIUM=recon, LOW=noise."""},
            {"role": "user", "content": log_entry},
        ],
    )
    return resp.choices[0].message.content
